drop table itemsinorder;
drop table weborders;

Create table weborders(
OrderID Int not null AUTO_INCREMENT,
UserID int,
Date date not null,
Firstname varchar(45),
Infix varchar(45),
Lastname varchar(45),
Postalcode varchar(45),
Housenumber varchar(45),
primary key(OrderID),
foreign key (UserID) references user(userID)
);

create table itemsInOrder(
OrderID int not null,
StockItemID int(11) not null,
Amount int not null,
primary key(OrderID, StockItemID),
foreign key(OrderID) references weborders(OrderID),
foreign key(StockItemID) references stockitems(StockItemID)
);
